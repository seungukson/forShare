#include "pch.h"
#include <stdio.h>

int GCD(int a, int b)
{
	if (b%a == 0)
		return a;
	return GCD(b%a, a);
}

int N;
const int SIZE = 100010;
int arr[SIZE];

int main()
{
	scanf("%d", &N);
	for (int i = 0; i < N; ++i)
	{
		scanf("%d", &arr[i]);
	}
	int mini = 1000100;
	int maxi = -1;

	for (int i = 1; i < N; ++i)
	{
		if (mini > arr[i] - arr[i - 1])
			mini = arr[i] - arr[i - 1];
		if (maxi < arr[i] - arr[i - 1])
			maxi = arr[i] - arr[i - 1];
	}
	int ans =GCD(mini, maxi);
	printf("%d",ans);

	return 0;

}
